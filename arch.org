- .xmonad/xmonad.hs
  #+BEGIN_SRC haskell
import System.IO
import XMonad
import XMonad.Config.Desktop
import XMonad.Hooks.DynamicLog
import XMonad.Util.EZConfig (additionalKeysP)
import XMonad.Util.Run (spawnPipe)
import XMonad.Layout.NoBorders

logHook' :: Handle -> X ()
logHook' xmobarProc = dynamicLogWithPP xmobarPP
                      { ppOutput = hPutStrLn xmobarProc
                      , ppCurrent = wrap "[" "]"
                      , ppTitle = shorten 50
                      , ppUrgent = xmobarColor "#dc322f" ""
                      }

main :: IO ()
main = do
  xmobarProc <- spawnPipe "xmobar"
  xmonad $ desktopConfig
       { modMask            = mod4Mask
       , terminal           = "alacritty"
       , focusedBorderColor = "#2aa198"
       , normalBorderColor  = "#002b36"
       , logHook            = logHook' xmobarProc
       , layoutHook         = smartBorders (layoutHook desktopConfig)
       }

       `additionalKeysP`
       [ ("M-b", spawn "firefox")
       , ("M-e", spawn "emacsclient -nc")
       , ("M-r", spawn "alacritty -e ranger")
       , ("M-s", spawn "flameshot gui")
       , ("M-p", spawn "dmenu_run -fn 'FantasqueSansMono Nerd Font' -nb '#002b36' -nf '#93a1a1' -sb '#859900' -sf '#002b36'")
       ]
#+END_SRC
